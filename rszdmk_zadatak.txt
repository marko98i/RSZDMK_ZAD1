char ocitaj_taster()
{
  char taster = 0;
  unsigned char row;
  
  MCUCR &= ~0x10;
  PORTD = 0x0f; 
  
  for(row = 0x80; row != 0x08; row >>= 1)
  {
    DDRD = row;
	_delay_ms(2);
    
    switch(row | ((~PIND) & 0x0f))
    {
      //case 0x88:
      	//taster = '1'; break;
      case 0x84:
      	taster = '2'; break;
      //case 0x82:
      //	taster = '3'; break;
     // case 0x81:
      //	taster = 'A'; break;
		
	  case 0x48:
      	taster = '4'; break;
      case 0x44:
      	taster = '5'; break;
      case 0x42:
      	taster = '6'; break;
     // case 0x41:
      //	taster = 'B'; break;
		
	 // case 0x28:
     // 	taster = '7'; break;
      case 0x24:
      	taster = '8'; break;
      //case 0x22:
      //	taster = '9'; break;
     // case 0x21:
      //	taster = 'C'; break;
		
	  case 0x18:
      	taster = '*'; break;
    //  case 0x14:
      //	taster = '0'; break;
     // case 0x12:
      //	taster = '#'; break;
     // case 0x11:
      //	taster = 'D'; break;
    }
  }
  
  return taster;
}

 int j=0,m;
